```typescript
export function useDriverDashboard() {
  return useQuery({
    queryKey: ['driver', 'dashboard'],
    queryFn: () => api.get('/delivery/stats').then((r) => r.data),
    staleTime: 30_000,
  });
}

export function useToggleOnline() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (isOnline: boolean) => api.put('/delivery/online', { isOnline }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['driver', 'dashboard'] });
      queryClient.invalidateQueries({ queryKey: ['driver', 'available'] });
    },
  });
}
```

**apps/mobile-delivery/hooks/useDriverDeliveries.ts:**
```typescript
export function useAvailableDeliveries() {
  return useQuery({
    queryKey: ['driver', 'available'],
    queryFn: () => api.get('/delivery/available').then((r) => r.data),
    refetchInterval: 30_000,               // Poll every 30s (less aggressive than vendor, driver is mobile)
    refetchIntervalInBackground: false,
  });
}

export function useCurrentDelivery() {
  return useQuery({
    queryKey: ['driver', 'current'],
    queryFn: async () => {
      try {
        const r = await api.get('/delivery/current');
        return r.data ?? null;
      } catch (e: any) {
        if (e?.response?.status === 404) return null;
        throw e;
      }
    },
    refetchInterval: 15_000,               // Poll current delivery status every 15s
    refetchIntervalInBackground: false,
  });
}

export function useAcceptDelivery() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (deliveryId: string) => api.post(`/delivery/${deliveryId}/accept`, {}),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['driver', 'available'] });
      queryClient.invalidateQueries({ queryKey: ['driver', 'current'] });
    },
  });
}

export function useUpdateDeliveryStatus() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ id, status }: { id: string; status: string }) =>
      api.put(`/delivery/${id}/status`, { status }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['driver', 'current'] });
      queryClient.invalidateQueries({ queryKey: ['driver', 'dashboard'] });
    },
  });
}
```

**apps/mobile-delivery/components/driver/SlideToConfirm.tsx (per D-12, Research Pattern 4):**
Uses `react-native-reanimated` and `react-native-gesture-handler` (both installed).
Props: `{ label: string; onConfirm: () => void; disabled?: boolean; color?: string }`.
Implementation:
- Track: full-width rounded pill (height 64, bg: color prop or '#FF6B35', opacity 0.15)
- Thumb: circle (56x56, same color, positioned absolute left)
- Label text centered in track: `"Slide to ${label}"`
- Use `useAnimatedGestureHandler` from reanimated v3 + `PanGestureHandler` from gesture-handler
- `translationX` clamped to `[0, trackWidth - thumbSize]`
- On gesture end: if `translationX > trackWidth * 0.75` → fire `onConfirm()` + haptic `Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success)` + spring-animate thumb to end + lock
- If below threshold: spring-animate back to 0 with `withSpring(0, { damping: 15 })`
- Disabled state: opacity 0.5, gestureHandler disabled

```typescript
// Key reanimated pattern:
const translateX = useSharedValue(0);
const gestureHandler = useAnimatedGestureHandler({
  onStart: (_, ctx: any) => { ctx.startX = translateX.value; },
  onActive: (event, ctx) => {
    translateX.value = Math.max(0, Math.min(ctx.startX + event.translationX, TRACK_WIDTH - THUMB_SIZE));
  },
  onEnd: () => {
    if (translateX.value > (TRACK_WIDTH - THUMB_SIZE) * 0.75) {
      translateX.value = withSpring(TRACK_WIDTH - THUMB_SIZE);
      runOnJS(onConfirm)();
    } else {
      translateX.value = withSpring(0, { damping: 15 });
    }
  },
});
```
Use `onLayout` on the track View to capture `TRACK_WIDTH` dynamically.

```typescript
import { Linking, Platform } from 'react-native';
function openNavigation(lat: number, lng: number, label: string) {
  const encoded = encodeURIComponent(label);
  const url = Platform.select({
    ios: `maps://?q=${encoded}&ll=${lat},${lng}`,
    android: `geo:${lat},${lng}?q=${lat},${lng}(${encoded})`,
  })!;
  Linking.canOpenURL(url).then((supported) => {
    if (supported) { Linking.openURL(url); }
    else { Linking.openURL(`https://maps.google.com/?q=${lat},${lng}`); }
  });
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| status update → API | Driver-controlled status transitions cross to API |
| Linking.openURL → native maps | Lat/lng from API is passed to native maps app |
| tel: links → phone dialer | Phone numbers from API open native dialer |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-05-01 | Tampering | PUT /delivery/:id/status | mitigate | API enforces valid state machine transitions server-side; client-side status map prevents invalid UI from appearing but server is authoritative |
| T-03-05-02 | Spoofing | POST /delivery/:id/accept | accept | API requires RequireDelivery middleware + isOnline check; cannot accept delivery assigned to another driver |
| T-03-05-03 | Tampering | Linking.openURL with lat/lng from API | accept | Coordinates come from authenticated API response; injected into maps:// URI — no user-controlled input in the URL |
| T-03-05-04 | Denial of Service | 15s polling on active delivery + 30s on available | accept | Low frequency, foreground-only; acceptable for delivery use case |
| T-03-05-05 | Information Disclosure | Customer phone shown on active delivery | accept | Customer phone is legitimately shared with driver for delivery coordination — this is a required feature (DRIV-03) |
</threat_model>
