```go
func (h *OrderHandler) TrackOrderWS(c *gin.Context) {
    orderID := c.Param("id")
    userID, ok := middleware.GetUserID(c)
    if !ok {
        c.JSON(http.StatusUnauthorized, gin.H{"error": "unauthorized"})
        return
    }

    // Verify the customer owns this order (security check — per D-13 trust boundary)
    var order models.Order
    if err := database.DB.Preload("Delivery").
        Where("id = ? AND customer_id = ?", orderID, userID).
        First(&order).Error; err != nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "order_not_found", "message": "Order not found"})
        return
    }
    if order.Delivery == nil || order.Delivery.ID == uuid.Nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "no_active_delivery", "message": "No active delivery for this order"})
        return
    }
    deliveryID := order.Delivery.ID.String()

    // Upgrade HTTP → WebSocket
    conn, err := wsUpgrader.Upgrade(c.Writer, c.Request, nil)
    if err != nil {
        log.Printf("WebSocket upgrade failed for order %s: %v", orderID, err)
        return // Upgrader writes error response
    }
    defer conn.Close()

    // Use a write channel to serialize WebSocket writes (gorilla/websocket is not concurrent-write-safe)
    writeCh := make(chan []byte, 32)
    defer close(writeCh)

    // Write pump goroutine — single goroutine owns conn.WriteMessage
    go func() {
        for msg := range writeCh {
            if err := conn.WriteMessage(websocket.TextMessage, msg); err != nil {
                log.Printf("WS write error for order %s: %v", orderID, err)
                return
            }
        }
    }()

    // Subscribe to core NATS (not JetStream) for location fan-out
    subject := fmt.Sprintf("%s.%s", services.SubjectDeliveryLocation, deliveryID)
    sub, err := services.GetNATSClient().Subscribe(subject, func(msg *nats.Msg) {
        select {
        case writeCh <- msg.Data:
        default:
            // Channel full — skip this message rather than block
        }
    })
    if err != nil {
        log.Printf("NATS subscribe failed for delivery %s: %v", deliveryID, err)
        conn.WriteMessage(websocket.CloseMessage, websocket.FormatCloseMessage(1011, "Internal error"))
        return
    }
    defer sub.Unsubscribe()

    // Read pump — blocks until client disconnects or error; handles pings and client close
    conn.SetReadLimit(512)
    for {
        if _, _, err := conn.ReadMessage(); err != nil {
            break
        }
    }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Mobile client → PUT /delivery/location | Driver app sends GPS coords; coords must be from a verified driver, not spoofed |
| Customer WebSocket → GET /orders/:id/track/ws | Customer connects to track order; must verify customer owns the order before upgrading |
| FCM token → push.go | Token provided by mobile client; must only be stored per authenticated user |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Spoofing | PUT /delivery/location | mitigate | Handler already requires auth via GIPAuth middleware; userID is extracted from verified JWT, not request body |
| T-04-02 | Denial of Service | PUT /delivery/location | mitigate | Per-user rate limiter (1 req/10s) added in Task 1; returns HTTP 429 on excess; prevents DB saturation |
| T-04-03 | Spoofing | GET /orders/:id/track/ws | mitigate | TrackOrderWS queries `WHERE id = ? AND customer_id = ?` — customer cannot track another customer's order |
| T-04-04 | Information Disclosure | WebSocket delivery.location.{id} | accept | Location data is real-time driver position, visible to the customer whose order it is; low sensitivity; accept |
| T-04-05 | Denial of Service | gorilla/websocket connection | mitigate | SetReadLimit(512) caps read size; write channel with non-blocking send (drops messages when full rather than blocking) |
| T-04-06 | Information Disclosure | push.go FCM token storage | accept | Token is device identifier only; no PII; registered per authenticated user; accept |
| T-04-07 | Elevation of Privilege | Vendor Accept/Reject via lock-screen | mitigate | Mobile app sends accept/reject to authenticated PUT /v1/chef/orders/:id/status; backend validates vendor owns the order before state change |
