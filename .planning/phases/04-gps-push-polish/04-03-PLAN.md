```typescript
async function setupPushNotifications(apiClient: /* your axios instance type */) {
  // Request push notification permissions
  const { status } = await Notifications.requestPermissionsAsync();
  if (status !== 'granted') return;

  // Android channels MUST be created before first notification (per Pitfall 8)

    await Notifications.setNotificationChannelAsync('order-updates', {

    });
    // VENDOR app channels (use this block instead for vendor):
    // await Notifications.setNotificationChannelAsync('new-orders', {
    //   name: 'New Orders',
    //   importance: Notifications.AndroidImportance.MAX,
    //   sound: 'default',
    //   vibrationPattern: [0, 250, 250, 250],
    // });
    // await Notifications.setNotificationChannelAsync('order-updates', { name: 'Order Updates', importance: Notifications.AndroidImportance.HIGH, sound: 'default' });
    // DRIVER app channels (use this block instead for driver):
    // await Notifications.setNotificationChannelAsync('new-deliveries', {
    //   name: 'New Deliveries',
    //   importance: Notifications.AndroidImportance.MAX,
    //   sound: 'default',
    //   vibrationPattern: [0, 250, 250, 250],
    // });
    // await Notifications.setNotificationChannelAsync('delivery-updates', { name: 'Delivery Updates', importance: Notifications.AndroidImportance.HIGH, sound: 'default' });
  }

  // Register FCM token
  const token = await getRawFCMToken();
  if (token) {
    await registerDeviceToken(apiClient, token);
  }

  // Listen for token rotation
  const tokenSub = Notifications.addPushTokenListener((event) => {
    registerDeviceToken(apiClient, event.data);
  });

  // Increment badge on foreground notification receipt
  const notifSub = Notifications.addNotificationReceivedListener(async () => {
    const current = await Notifications.getBadgeCountAsync();
    await Notifications.setBadgeCountAsync(current + 1);
  });

  return () => {
    tokenSub.remove();
    notifSub.remove();
  };
}
```

For the vendor app, the Accept/Reject action MUST use `SecureStore.getItemAsync` + `fetch` directly — NOT the React-context-bound axios instance. The notification response listener can fire when the app is backgrounded or killed (no React context), and axios instances bound to React providers will silently fail with no auth header.

  Use this exact pattern for vendor Accept/Reject:
  ```typescript
  import * as SecureStore from 'expo-secure-store';

  // Inside addNotificationResponseReceivedListener:
  if (actionId === 'ACCEPT_ORDER' || actionId === 'REJECT_ORDER') {
    const token = await SecureStore.getItemAsync('auth_token'); // use your app's SecureStore key
    if (!token || !data?.orderId) return;
    const status = actionId === 'ACCEPT_ORDER' ? 'accepted' : 'rejected';
    await fetch(`${API_BASE_URL}/v1/chef/orders/${data.orderId}/status`, {
      method: 'PUT',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ status }),
    });
    return; // do not navigate
  }
  ```

  Do NOT use `apiClient.put(...)` for lock-screen actions. Do NOT use any imported axios instance. SecureStore+fetch is the only approach that works reliably from a backgrounded or killed app context (per RESEARCH.md Pitfall 2 — background task API calls cannot use React context).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Notification action → vendor API call | Lock-screen Accept/Reject calls order status endpoint; must verify vendor owns the order |
| FCM token storage | Token stored per user in DB; must only be queried for that user's notifications |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-13 | Spoofing | Vendor lock-screen Accept/Reject | mitigate | API call in notification action uses authenticated API client (Bearer JWT); backend verifies chef_id = current user on order status change |
| T-04-14 | Information Disclosure | Push notification content | mitigate | Notification body contains no PII; only order status and UUID; orderId in data payload is a UUID (not customer name/address) |
| T-04-15 | Denial of Service | NATS push consumer | mitigate | Push calls wrapped in goroutines; FCM failures logged but do not crash the consumer loop; consumer continues processing next event |
| T-04-16 | Privacy | FCM token in DB | accept | FCM token is a device identifier registered per authenticated user; no PII; stored against user record; accept |
| T-04-17 | Elevation of Privilege | opensAppToForeground: false | mitigate | Accept/Reject action does not open app — API call happens in background; orderId verified server-side; chef cannot accept another chef's order |
</threat_model>
