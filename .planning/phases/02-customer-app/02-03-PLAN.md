**hooks/useOrders.ts**:

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '../lib/api';
import type { Order } from '../types/customer';

interface CreateOrderPayload {
  chefId: string;
  items: { menuItemId: string; quantity: number }[];
  deliveryAddressId: string;
  note?: string;
}

export function useCreateOrder() {
  const queryClient = useQueryClient();
  return useMutation<{ data: Order }, Error, CreateOrderPayload>({
    mutationFn: (payload) => api.post('/v1/orders', payload).then((r) => r.data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['orders'] });
    },
  });
}

export function useOrders(params: { page?: number; limit?: number; status?: string } = {}) {
  return useQuery<{ data: Order[]; meta: { total: number; page: number; limit: number } }>({
    queryKey: ['orders', params],
    queryFn: () => api.get('/v1/orders', { params }).then((r) => r.data),
    staleTime: 1000 * 30, // 30 seconds
  });
}

export function useOrder(id: string) {
  return useQuery<{ data: Order }>({
    queryKey: ['order', id],
    queryFn: () => api.get(`/v1/orders/${id}`).then((r) => r.data),
    enabled: !!id,
  });
}

// Used for polling after payment — stops when order status shows payment confirmed
export function useOrderStatus(id: string, enabled: boolean) {
  return useQuery<{ data: Order }>({
    queryKey: ['order-status', id],
    queryFn: () => api.get(`/v1/orders/${id}`).then((r) => r.data),
    enabled: enabled && !!id,
    refetchInterval: (query) => {
      const status = query.state.data?.data?.status;
      // Stop polling when payment is confirmed (status moves past pending)
      if (status && status !== 'pending') return false;
      return 3000; // 3 seconds
    },
    refetchIntervalInBackground: false,
  });
}
```

**Place Order flow** (implement as `handlePlaceOrder` async function):

```typescript
async function handlePlaceOrder() {
  setIsLoading(true);
  setError(null);
  
  try {
    // Step 1: Create order
    const { data: order } = await createOrder.mutateAsync({
      chefId: cartStore.chefId!,
      items: cartStore.items.map((i) => ({ menuItemId: i.menuItemId, quantity: i.quantity })),
      deliveryAddressId: selectedAddressId,
      note: note.trim() || undefined,
    });

    // Step 2: Create Razorpay payment order
    const { data: paymentData } = await api.post(`/v1/payments/order/${order.data.id}/create`, {});

    // Step 3: Build hosted checkout URL (D-04 fallback — expo-web-browser)
    const checkoutUrl = [
      'https://api.razorpay.com/v1/checkout/embedded',
      `?key_id=${paymentData.razorpayKeyId}`,
      `&order_id=${paymentData.razorpayOrderId}`,
      `&amount=${paymentData.amount}`,
      `&currency=${paymentData.currency}`,
      `&name=Fe3dr`,
      `&prefill[name]=${encodeURIComponent(paymentData.prefill?.name ?? '')}`,
      `&prefill[email]=${encodeURIComponent(paymentData.prefill?.email ?? '')}`,
      `&prefill[contact]=${encodeURIComponent(paymentData.prefill?.phone ?? '')}`,
      `&callback_url=${encodeURIComponent('homechef-customer://payment/result')}`,
    ].join('');

    // Step 4: Open in-app browser
    await WebBrowser.openBrowserAsync(checkoutUrl);

    // Step 5: Start polling order status (webhook updates server-side)
    // setPollingOrderId triggers useOrderStatus hook to start polling
    setPollingOrderId(order.data.id);
    
  } catch (err) {
    setError('Order creation failed. Please try again.');
    setIsLoading(false);
  }
}
```

**Polling logic** (in the same checkout.tsx component):

```typescript
const { data: polledOrder } = useOrderStatus(pollingOrderId ?? '', !!pollingOrderId);

useEffect(() => {
  if (!polledOrder) return;
  const status = polledOrder.data.status;
  if (status !== 'pending') {
    // Payment confirmed or failed
    setPollingOrderId(null);
    setIsLoading(false);
    if (status === 'confirmed' || status === 'preparing') {
      useCartStore.getState().clearCart();
      Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
      router.replace(`/order/${polledOrder.data.id}`);
    } else if (status === 'cancelled') {
      setError('Payment was not completed. Please try again.');
    }
  }
}, [polledOrder]);

// Timeout: stop polling after 60s regardless
useEffect(() => {
  if (!pollingOrderId) return;
  const timeout = setTimeout(() => {
    setPollingOrderId(null);
    setIsLoading(false);
    setError('Payment confirmation timed out. Check your order history to confirm status.');
  }, 60000);
  return () => clearTimeout(timeout);
}, [pollingOrderId]);
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Cart → POST /v1/orders | Cart item prices and chefId cross to Go API for order creation |
| App → Razorpay hosted page | User enters payment details inside Razorpay iframe in browser — not in app |
| Razorpay webhook → Go API | Payment status updated server-side by webhook, not by client |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03-01 | Tampering | Cart prices sent to POST /v1/orders | mitigate | Go orders.go validates menu item prices server-side; client-side prices are display only — backend is source of truth |
| T-02-03-02 | Spoofing | Razorpay callback_url deep link | accept | Deep link only navigates back to checkout; payment status determined by server-side webhook + polling, not by URL params |
| T-02-03-03 | Information Disclosure | razorpayKeyId in checkout URL | accept | razorpayKeyId is the public API key (not secret); safe to expose in URL. Secret key never leaves server (payment.go verified) |
| T-02-03-04 | Tampering | pollingOrderId in component state | accept | Client polls GET /v1/orders/:id; Go API validates JWT and returns only the authenticated user's order |
| T-02-03-05 | Denial of Service | Infinite polling loop | mitigate | 60-second hard timeout kills polling; status check stops on any non-pending status |
</threat_model>
