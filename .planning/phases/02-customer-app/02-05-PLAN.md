```typescript
import { useQuery } from '@tanstack/react-query';
import { api } from '../lib/api';
import type { Order } from '../types/customer';

export function useOrders(params: { page?: number; limit?: number; status?: string } = {}) {
  return useQuery<{ data: Order[]; meta: { total: number; page: number; limit: number } }>({
    queryKey: ['orders', params],
    queryFn: () => api.get('/v1/orders', { params }).then((r) => r.data),
    staleTime: 1000 * 30, // 30 seconds
  });
}

export function useOrder(id: string) {
  return useQuery<{ data: Order }>({
    queryKey: ['order', id],
    queryFn: () => api.get(`/v1/orders/${id}`).then((r) => r.data),
    enabled: !!id,
  });
}
```

**hooks/useFavorites.ts**:

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '../lib/api';
import type { Chef } from '../types/customer';

export function useFavorites() {
  return useQuery<{ data: Chef[] }>({
    queryKey: ['favorites'],
    queryFn: () => api.get('/v1/favorites').then((r) => r.data),
    staleTime: 1000 * 60 * 2,
  });
}

export function useToggleFavorite() {
  const queryClient = useQueryClient();
  return useMutation<void, Error, { chefId: string; isFavorited: boolean }>({
    mutationFn: ({ chefId, isFavorited }) =>
      isFavorited
        ? api.delete(`/v1/favorites/chefs/${chefId}`).then(() => undefined)
        : api.post(`/v1/favorites/chefs/${chefId}`, {}).then(() => undefined),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['favorites'] });
      queryClient.invalidateQueries({ queryKey: ['chefs'] }); // update isFavorited on browse screen
    },
  });
}
```

**hooks/useProfile.ts**:

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '../lib/api';

interface Profile {
  id: string;
  firstName: string;
  lastName: string;
  email: string;
  phone?: string;
  cuisinePreferences?: string[];
}

export function useProfile() {
  return useQuery<{ data: Profile }>({
    queryKey: ['profile'],
    queryFn: () => api.get('/v1/profile').then((r) => r.data),
    staleTime: 1000 * 60 * 5,
  });
}

export function useUpdateProfile() {
  const queryClient = useQueryClient();
  return useMutation<{ data: Profile }, Error, Partial<Profile>>({
    mutationFn: (payload) => api.patch('/v1/profile', payload).then((r) => r.data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['profile'] });
    },
  });
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Profile form → PATCH /v1/profile | User-controlled name, phone, preferences cross to Go API |
| Catering form → POST /v1/catering/requests | Event details (date, location, budget) cross to Go API |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-05-01 | Tampering | Profile update form | mitigate | Zod schema validates firstName/lastName min length, phone regex, cuisinePreferences as string array |
| T-02-05-02 | Tampering | Catering request form | mitigate | Zod schema validates required fields (date, guestCount as positive integer, budget as positive number) before API call |
| T-02-05-03 | Spoofing | Favorites toggle optimistic update | accept | Optimistic removal is cosmetic; on error, data is restored by React Query. No financial or security impact |
| T-02-05-04 | Information Disclosure | Order history list | mitigate | useOrders fetches via JWT-authenticated API call; Go API scopes results to the requesting user |
| T-02-05-05 | Denial of Service | Social feed infinite scroll | mitigate | Use page-based pagination (limit=20); React Query caches pages; do not load all posts at once |
</threat_model>
