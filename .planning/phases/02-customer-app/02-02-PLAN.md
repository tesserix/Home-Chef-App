**Step B — hooks/useChefs.ts**:

Create hooks for chef listing, single chef detail, and chef menu. Reference apps/api/handlers/chefs.go for confirmed query params.

```typescript
// apps/mobile-customer/hooks/useChefs.ts
import { useQuery } from '@tanstack/react-query';
import { api } from '../lib/api';
import type { Chef, MenuItem } from '../types/customer';

export interface ChefFilters {
  search?: string;
  cuisine?: string;
  dietary?: string;
  rating?: number;
  isOpen?: boolean;
  sort?: 'rating' | 'orders' | 'newest' | 'price';
  page?: number;
  limit?: number;
}

export function useChefs(filters: ChefFilters = {}) {
  return useQuery<{ data: Chef[]; meta: { total: number; page: number; limit: number } }>({
    queryKey: ['chefs', filters],
    queryFn: () => api.get('/v1/chefs', { params: filters }).then((r) => r.data),
    staleTime: 1000 * 60 * 2, // 2 minutes — chef list changes infrequently
  });
}

export function useChef(id: string) {
  return useQuery<{ data: Chef }>({
    queryKey: ['chef', id],
    queryFn: () => api.get(`/v1/chefs/${id}`).then((r) => r.data),
    enabled: !!id,
  });
}

export function useChefMenu(chefId: string) {
  return useQuery<{ data: MenuItem[] }>({
    queryKey: ['chef-menu', chefId],
    queryFn: () => api.get(`/v1/chefs/${chefId}/menu`).then((r) => r.data),
    enabled: !!chefId,
    staleTime: 1000 * 60 * 5, // 5 minutes — menu changes rarely
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| filter state → API params | User-controlled filter values sent as query params to Go API |
| Cart store → Checkout | Cart contents drive order creation in next plan |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-01 | Tampering | Chef filter params in useChefs | accept | Go API performs server-side validation; client-side params are display convenience only |
| T-02-02-02 | Tampering | Cart store (client-side) | mitigate | Prices are stored from API response at add-time and re-validated by Go API during order creation (orders.go validates item prices) |
| T-02-02-03 | Information Disclosure | Chef photo URLs in expo-image | accept | Images are public CDN URLs; no authentication required |
| T-02-02-04 | Denial of Service | 2-column FlatList with large dataset | mitigate | FlatList is virtualized (only renders visible items); use `limit=20` default in useChefs filters |
</threat_model>
