**Step A — hooks/useOrderTracking.ts** (per D-05 — 5s polling, auto-stop):

Implement exactly as per RESEARCH.md Code Examples section. Key requirements:
- `refetchInterval` is a FUNCTION (not a constant) that checks `query.state.data?.data?.status`
- Returns `false` when status is `'delivered'` or `'cancelled'` to stop polling
- Returns `5000` for all other statuses
- `refetchIntervalInBackground: false` — do NOT poll when app is backgrounded
- `enabled: enabled && !!orderId` — only start when both conditions true

```typescript
// apps/mobile-customer/hooks/useOrderTracking.ts
import { useQuery } from '@tanstack/react-query';
import { api } from '../lib/api';
import type { TrackingResponse } from '../types/customer';

export function useOrderTracking(orderId: string, enabled: boolean = true) {
  return useQuery<{ data: TrackingResponse }>({
    queryKey: ['order-tracking', orderId],
    queryFn: () => api.get(`/v1/orders/${orderId}/track`).then((r) => r.data),
    enabled: enabled && !!orderId,
    refetchInterval: (query) => {
      const status = query.state.data?.data?.status;
      if (status === 'delivered' || status === 'cancelled') return false;
      return 5000; // D-05: 5 second polling
    },
    refetchIntervalInBackground: false,
    staleTime: 4000, // just under polling interval to always show fresh data
  });
}
```

**Step B — components/tracking/DeliveryMap.tsx** (per D-07, D-06):

Full-screen MapView with two markers. PROVIDER_DEFAULT (not PROVIDER_GOOGLE) unless `process.env.EXPO_PUBLIC_GOOGLE_MAPS_KEY` is set — this avoids requiring a Google Maps API key for v1 builds.

```typescript
// apps/mobile-customer/components/tracking/DeliveryMap.tsx
import MapView, { Marker, PROVIDER_DEFAULT } from 'react-native-maps';
import { StyleSheet } from 'react-native';

interface DeliveryMapProps {
  driverLat?: number | null;
  driverLng?: number | null;
  destinationLat: number;
  destinationLng: number;
  chefLat?: number;
  chefLng?: number;
}

export function DeliveryMap({
  driverLat,
  driverLng,
  destinationLat,
  destinationLng,
  chefLat,
  chefLng,
}: DeliveryMapProps) {
  // Guard per Pitfall 6: driver may have 0,0 before first location update
  const hasDriverLocation = driverLat && driverLng && driverLat !== 0 && driverLng !== 0;
  
  // Initial region centers on destination; if driver has location, center between both
  const initialRegion = {
    latitude: destinationLat,
    longitude: destinationLng,
    latitudeDelta: 0.05,
    longitudeDelta: 0.05,
  };

  return (
    <MapView
      style={StyleSheet.absoluteFillObject}
      provider={PROVIDER_DEFAULT}
      initialRegion={initialRegion}
      showsUserLocation={false}
      showsMyLocationButton={false}
    >
      {/* Delivery destination marker (red) */}
      <Marker
        coordinate={{ latitude: destinationLat, longitude: destinationLng }}
        title="Delivery Address"
        pinColor="red"
      />
      {/* Driver location marker (blue) — only when driver has real coordinates */}
      {hasDriverLocation && (
        <Marker
          coordinate={{ latitude: driverLat!, longitude: driverLng! }}
          title="Driver"
          pinColor="blue"
        />
      )}
      {/* Chef/pickup marker (orange) — show while waiting for pickup */}
      {chefLat && chefLng && (
        <Marker
          coordinate={{ latitude: chefLat, longitude: chefLng }}
          title="Pickup Point"
          pinColor="orange"
        />
      )}
    </MapView>
  );
}
```

**Step A — components/orders/OrderTimeline.tsx**:

Visual status progression timeline. Four ordered steps matching the order lifecycle:

```
Confirmed → Preparing → Picked Up → Delivered
```

Each step shows: icon, label, timestamp (if available), and completion state (grey = pending, primary color = complete, animated highlight = current).

Props: `status: Order['status']`, `estimatedDeliveryTime?: string`.

Step completion logic:
```typescript
const STEPS = ['confirmed', 'preparing', 'picked_up', 'delivered'] as const;
const currentIndex = STEPS.indexOf(status as typeof STEPS[number]);
// stepIndex <= currentIndex → completed
// stepIndex === currentIndex + 1 → active (in-progress)
// stepIndex > currentIndex + 1 → pending
```

Use NativeWind for styling. Completed steps use `text-green-600` and filled circle; current step uses `text-primary`; pending steps use `text-gray-400`.

Full-screen stack screen with:

2. @gorhom/bottom-sheet overlaid on top (snapPoints: ['30%', '60%'])

Screen skeleton:
```typescript
import { View } from 'react-native';
import { useLocalSearchParams } from 'expo-router';
import { useIsFocused } from '@react-navigation/native';  // or expo-router equivalent
import { useOrderTracking } from '../../hooks/useOrderTracking';
import { DeliveryMap } from '../../components/tracking/DeliveryMap';
import { OrderTimeline } from '../../components/orders/OrderTimeline';
import BottomSheet from '@gorhom/bottom-sheet';
import { useRef, useMemo, useCallback } from 'react';
import { router } from 'expo-router';

export default function TrackOrderScreen() {
  const { id } = useLocalSearchParams<{ id: string }>();
  const isFocused = useIsFocused();  // per Anti-Patterns: guard polling with focus
  
  const { data, isLoading } = useOrderTracking(id, isFocused);
  const tracking = data?.data;
  
  const bottomSheetRef = useRef<BottomSheet>(null);
  const snapPoints = useMemo(() => ['30%', '60%'], []);
  
  // Navigate to order detail when delivery is complete
  useEffect(() => {
    if (tracking?.status === 'delivered') {
      setTimeout(() => router.replace(`/order/${id}`), 2000);
    }
  }, [tracking?.status]);

  if (isLoading || !tracking) {
    return <View className="flex-1 items-center justify-center bg-background"><ActivityIndicator /></View>;
  }

  return (
    <View style={{ flex: 1 }}>
      <DeliveryMap
        driverLat={tracking.delivery?.currentLatitude}
        driverLng={tracking.delivery?.currentLongitude}
        destinationLat={/* delivery address lat from tracking order */}
        destinationLng={/* delivery address lng from tracking order */}
        chefLat={tracking.chef?.latitude}
        chefLng={tracking.chef?.longitude}
      />
      <BottomSheet ref={bottomSheetRef} snapPoints={snapPoints} index={0}>
        <View className="px-4 py-2">
          <Text className="text-text-primary font-semibold text-base">
            Order #{tracking.orderNumber}
          </Text>
          {tracking.estimatedDeliveryTime && (
            <Text className="text-text-secondary text-sm mt-1">
              Estimated arrival: {new Date(tracking.estimatedDeliveryTime).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
            </Text>
          )}
          <OrderTimeline status={tracking.status} estimatedDeliveryTime={tracking.estimatedDeliveryTime} />
        </View>
      </BottomSheet>
    </View>
  );
}
```

NOTE: The tracking response does not include the delivery destination coordinates directly — it includes the delivery address. The coordinates for the destination marker should come from:
1. The order's `deliveryAddress` (if the address has lat/lng stored) OR
2. Default to `0, 0` as fallback coordinates — show a note "Location unavailable" if 0,0

Read the actual TrackingResponse carefully: if no lat/lng in the delivery address, use the chef's coordinates as a safe fallback for the destination marker until delivery address geocoding is available in a future phase.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04-01 | Information Disclosure | Driver GPS coordinates | accept | Coordinates are only shown to the authenticated customer who owns that order (Go API validates JWT + order ownership) |
| T-02-04-02 | Denial of Service | 5s polling on tracking screen | mitigate | useIsFocused guard stops polling when screen is not active; refetchIntervalInBackground:false stops polling when app backgrounds |
